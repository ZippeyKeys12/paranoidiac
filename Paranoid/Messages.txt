//Map02: Message 1
class MessageItem01 : CustomInventory
{
	Default
	{
		Tag "Message";
		Scale .25;
		Inventory.Icon "MAIL01";
		inventory.usesound "misc/chat";
		+INVENTORY.INVBAR
		+INVENTORY.UNDROPPABLE
	}
	States
	{
	Spawn:
		MAIL A -1;
		Stop;
	Use:
		TNT1 A 0 ACS_NamedExecuteAlways("HintMessage",0,1);
		Fail;
	}
}
