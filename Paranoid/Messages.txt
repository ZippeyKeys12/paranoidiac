class MessageItem : CustomInventory
{
	override bool TryPickup(in out Actor toucher)
	{
		toucher.A_PlaySound("radiostatic",6);
		toucher.ACS_NamedExecuteAlways("HintMessage");
		return Super.TryPickup(toucher);
	}
	int MessageNum;
	Property MessageNum:MessageNum;
	Default
	{
		Tag "Message";
		Inventory.UseSound "";
		+INVENTORY.INVBAR
		+INVENTORY.UNDROPPABLE
	}
	States
	{
	Use:
		TNT1 A 0 ACS_NamedExecuteAlways("HintMessage",0,invoker.MessageNum);
		Fail;
	}
}

//P1M2: Message 1
class MessageItemP1M2VOX1 : MessageItem
{
	Default
	{
		Inventory.Icon "MAIL01";
		MessageItem.MessageNum 1;
	}
}
