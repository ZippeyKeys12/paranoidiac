class HLBabyHeadCrab : HLHeadcrab
{
	Default
	{
		Height 8;
		-Solid
		RenderStyle "Translucent";
		Alpha .9;
		MaxTargetRange 128;
		Scale .09;
		maxdropoffheight 24;
		Species "HLBabyHeadCrab";
	}
	States
	{
	Spawn:
		HLQI A 3 A_Look;
		Loop;
	See:
		TNT1 A 0 A_JumpIf(crabjump,"SeeMelee");
		HLQW A 1 A_Chase(null,"Missile");
		HLQW AAAA 1 A_Chase(null,null,HLSMOOTHCHASE);
		HLQW B 1 A_Chase(null,"Missile");
		HLQW BBBB 1 A_Chase(null,null,HLSMOOTHCHASE);
	SeeCont:
		HLQW C 1 A_Chase(null,"Missile");
		HLQW CCCC 1 A_Chase(null,null,HLSMOOTHCHASE);
		HLQW D 1 A_Chase(null,"Missile");
		HLQW DDDD 1 A_Chase(null,null,HLSMOOTHCHASE);
		HLQW E 1 A_Chase(null,"Missile");
		HLQW EEEE 1 A_Chase(null,null,HLSMOOTHCHASE);
		Loop;
	SeeMelee:
		HLQW A 1 A_Chase("Melee",null);
		HLQW AAAA 1 A_Chase(null,null,HLSMOOTHCHASE);
		HLQW B 1
		{
			crabjump=0;
			A_Chase("Melee",null);
		}
		HLQW BBBB 1 A_Chase(null,null,HLSMOOTHCHASE);
		Goto SeeCont;
	Melee:
		HLQW A 10 A_CustomMeleeAttack(random(1,8),"halflife/monsters/headcrab/meleehit");
		Goto See;
	Missile:
		HLQI A 4 A_FaceTarget;
		HLQM AB 3;
		HLQM C 1
		{
			crabjump=1;
			A_SkullAttack();
		}
		HLQM C 5 AddZ(random(10,25)/4.0);
		HLQM C 3 A_CheckFloor(1);
		Wait;
		HLQM D 60 A_stop;
		Goto See;
	Pain:
		HLQP A 8 {crabjump=0;}
		Goto See;
	Death:
	XDeath:
	  	TNT1 AAAA 0 A_SpawnItemEx("XenGibsMini",random(2,-2),random(2,-2),random(0,20),random(1,4),random(1,4),random(2,6),random(0,359));
		Stop;
	}
}
