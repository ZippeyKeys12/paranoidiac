class ParanoidAmmoHandler : StaticEventHandler
{
	override void WorldTick()
	{
		if(level.mapname=="TITLEMAP")
			return;
		for(int i=0;i<MAXPLAYERS;i++)
		{
			if(playeringame[i])
			{
				let me=players[i].mo;
				int should=me.CountInv("M16HeldAmmo")+me.CountInv("M16Gren");
				if(me.CountInv("M16ClipGren")!=should)
					me.SetInventory("M16ClipGren",should);
			}
		}
	}
}
