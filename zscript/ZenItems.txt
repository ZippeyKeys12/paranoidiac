class ZenTree : Actor
{
	Default
	{
		Height 144;
		Health 10000;
		Mass 0x7FFFFFFF;
		meleerange 128;
		PainChance 128;
		PainSound "tree/pain";
		+NOTAUTOAIMED
		+SHOOTABLE
		+NOBLOOD
		+FLOORCLIP
		+NORADIUSDMG
		+NODAMAGE
		+SOLID
		+FLOORCLIP
	}
	States
	{
	Spawn:
		POSS A 8 A_LookEx(LOF_NOSOUNDCHECK,0,136,136,72,"See");
		Loop;
	See:
		POSS A 8 A_Chase;
		Goto Spawn;
	Melee:
		POSS BC 3;
		POSS DEF 1 A_FaceTarget;
		POSS G 1 A_CustomMeleeAttack(10,"tree/attack","tree/swipe");
		POSS HI 4;
		Goto Spawn;
	Pain:
		POSS BC 1 A_Pain;
		POSS D 6;
		POSS CB 3;
		Goto Spawn;
	}
}
