ACTOR ZenLight 30283
{
	Radius 8
	Height 32
	Health 10000
	Mass 0x7FFFFFFF
	PainChance 192
	PainSound "misc/chat"
	+SHOOTABLE
	+NOTAUTOAIMED
	+NOBLOOD
	+FLOORCLIP
	+NORADIUSDMG
	+NODAMAGE
	+SOLID
	+RANDOMIZE
	States
	{
	Spawn:
		POSS A 24 Bright
		POSS ABCDEFGHIJK 6 Bright A_LookEx (LOF_NOSOUNDCHECK, 0, 264, 264, 45, "Shrink")
		POSS A 0 A_JumpIfCloser (256, "Shrink")
		Goto Spawn+1
	Shrink:
		POSS LMNO 3 Bright
		POSS P 3
	Shrunk1:
		POSS Q 8
		POSS Q 0 A_JumpIfCloser(256,"Shrunk1")
		goto Grow
	Pain:
		POSS Q 24
		goto Shrunk1
	Grow:
		POSS Q 70
		POSS P 3
		POSS ONML 3 Bright
		goto Spawn
	}
}

ACTOR ZenTree 30284
{
	Height 144
	Health 10000
	Mass 0x7FFFFFFF
	meleerange 128
	PainChance 128
	PainSound "tree/pain"
	+NOTAUTOAIMED
	+SHOOTABLE
	+NOBLOOD
	+FLOORCLIP
	+NORADIUSDMG
	+NODAMAGE
	+SOLID
	+FLOORCLIP
	States
	{
	Spawn:
		POSS A 8 A_LookEx (LOF_NOSOUNDCHECK, 0, 136, 136, 72, "See")
		Loop
	See:
		POSS A 8 A_Chase
		goto Spawn
	Melee:
		POSS BC 3
		POSS DEF 1 A_FaceTarget
		POSS G 1 A_CustomMeleeAttack (10, "tree/attack", "tree/swipe")
		POSS HI 4
		goto Spawn
	Pain:
		POSS BC 1 A_Pain
		POSS D 6
		POSS CB 3
		goto Spawn
	}
}
