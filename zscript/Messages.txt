{
	Default
	{
		Tag "Message";
		Scale .25;
		Inventory.Icon "MAIL01";
		inventory.usesound "misc/chat";
		+INVENTORY.INVBAR
		+INVENTORY.UNDROPPABLE
	}
	States
	{
	Spawn:
		MAIL A -1;
		Stop;
	Use:
		TNT1 A 0 ACS_Execute(501,0);
		Fail;
	}
