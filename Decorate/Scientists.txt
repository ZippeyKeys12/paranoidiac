actor HLScientist1Panic : HLScientist1Wander  4043
{
   Monster
   Health 25
   Speed 5
   Height 56
   Radius 20
   PainChance 255
   Scale 0.36
   -NODAMAGE
   +FRIENDLY
   +FRIGHTENED
   -CANUSEWALLS
   -COUNTKILL
   Activation THINGSPEC_NoDeathSpecial
   SeeSound "halflife/monsters/scientist/sight"
   PainSound "halflife/monsters/scientist/pain"
   DeathSound "halflife/monsters/scientist/death"
//   ActiveSound "halflife/monsters/scientist/idle"
   States
   {
   Spawn:
      TNT1 A 0 
      TNT1 A 0 A_ClearTarget
      TNT1 A 0 A_Jump(255,"Wander","WanderRunStart")
   Wander:
      HLRW AA 3 A_Wander
      TNT1 a 0 A_Look
      HLRW BB 3 A_Wander
      TNT1 a 0 A_Look
      HLRW CC 3 A_Wander
      TNT1 a 0 A_Look
      HLRW DD 3 A_Wander
      TNT1 a 0 A_Look
      HLRW EE 3 A_Wander
      TNT1 a 0 A_Look
      HLRW FF 3 A_Wander
      TNT1 a 0 A_Look
      TNT1 A 0 A_Jump(8,"SayWhileMoving")   //how often they'll say something while moving
      TNT1 A 0 A_Jump(4,"StopMov")			//Reduced from 32 to 4
      Loop
   StopMov:
      TNT1 A 0 A_Jump(235,"Pond1","Pond2","IdleReg")   //stopmov, ironically also randomly starts the run seq
      TNT1 A 0 A_Jump(255,"WanderRunStart")      //kind of rare to see them running just because of nothing.
   SayWhileMoving:
      TNT1 A 0 A_PlaySound("scientist/panic")
      Goto "Wander"
   IdleReg:
      TNT1 A 0 A_Jump(3,"SayPhrase1")
      HLRI AAA 5 A_Look
      TNT1 A 0 A_Jump(5,"Wander")
      Loop
   Pond1:
      TNT1 A 0 A_Jump(2,"SayPhrase2")
      HLRI BBB 5 A_Look
      TNT1 A 0 A_Jump(6,"Wander")
      Loop
   Pond2:
      TNT1 A 0 A_Jump(2,"SayPhrase3")
      HLRI CCC 5 A_Look
      TNT1 A 0 A_Jump(6,"Wander")
      Loop
   SayPhrase1:
      TNT1 A 0 A_PlaySound("scientist/panic")
      Goto "IdleReg"+1
   SayPhrase2:
      TNT1 A 0 A_PlaySound("scientist/panic")
      Goto "Pond1"+1
   SayPhrase3:
      TNT1 A 0 A_PlaySound("scientist/panic")
      Goto "Pond2"+1
   See:            //scream the 1st time they enter here
      TNT1 A 0 A_PlaySound("halflife/monsters/scientist/sight")
   SeeDecide:         //decide if they calm down or not
      TNT1 A 0 A_JumpIfInventory("PanicCounter",35,"SeeCalmDown")
   SeeLoop:
      HLRR AAABBBCCCDDDEEEFFF 1 A_Chase
      TNT1 A 0 A_GiveInventory("PanicCounter",1)
      TNT1 A 0 A_Jump(4,"SeeOMGsnd")   //every once in a while it will say something
      Goto "SeeDecide"
   SeeCalmDown:
      TNT1 A 0 A_TakeInventory("PanicCounter",8)   //If decides to go running againg, will last a bit more
      TNT1 A 0 A_Jump(160,"WanderRun")
      TNT1 A 0 A_TakeInventory("PanicCounter",100)
      TNT1 A 0 A_ClearTarget
      Goto "Wander"
   SeeOMGsnd:
      TNT1 A 0 A_PlaySound("halflife/monsters/scientist/sight")   // OMG OMG OMG!
      Goto "SeeLoop"

   WanderRunStart:      //scream the 1st time they enter here
      TNT1 A 0 A_PlaySound("halflife/monsters/scientist/sight")
   WanderRunDecide:   //decide if they calm down or not
      TNT1 A 0 A_JumpIfInventory("PanicCounter",30,"CalmDown")
   WanderRun:         //not calmed at all
      HLRR AAA 1 A_Wander
      TNT1 a 0 A_Look
      HLRR BBB 1 A_Wander
      TNT1 a 0 A_Look
      HLRR CCC 1 A_Wander
      TNT1 a 0 A_Look
      HLRR DDD 1 A_Wander
      TNT1 a 0 A_Look
      HLRR EEE 1 A_Wander
      TNT1 a 0 A_Look
      HLRR FFF 1 A_Wander
      TNT1 a 0 A_Look
      TNT1 A 0 A_GiveInventory("PanicCounter",1)
      TNT1 A 0 A_Jump(4,"OMGsnd")   //every once in a while it will say something
      Goto "WanderRunDecide"
   OMGsnd:
      TNT1 A 0 A_PlaySound("halflife/monsters/scientist/sight")   // OMG OMG OMG!
      Goto "WanderRun"   
   CalmDown:
      TNT1 A 0 A_TakeInventory("PanicCounter",8)   //If decides to go running again, will last a bit more
      TNT1 A 0 A_Jump(160,"WanderRun")
      TNT1 A 0 A_TakeInventory("PanicCounter",100)
      Goto "Wander"
   Pain:
      HLRP A 8 A_Pain
      Goto "See"
   Death:
      HLRD A 3 A_Scream
      HLRD B 3 A_Fall
      HLRD CDEFGHIJ 3
      TNT1 A 0 A_PlaySound("halflife/weapons/bodydrop")
      HLRD KL 3
      HLRD M -1
      Stop
   }
}
